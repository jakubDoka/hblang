loops 1.fib:
	push rbx
	mov rsi, rdi
	mov rax, 0x00
	mov rcx, 0x01
	mov rdx, rax
	mov rax, rsi
1:	cmp rax, 0x00
	setz bl
	movzx rbx, bl
	test bl, bl
	jz :0
	mov rax, rdx
	pop rbx
	ret
0:	add rdx, rcx
	sub rax, 0x01
	xchg rcx, rdx
	jmp :1
main:
	sub rsp, 0x08
	mov rsi, 0x0A
	mov rdi, rsi
	call :loops 1.fib
	add rsp, 0x08
	ret
