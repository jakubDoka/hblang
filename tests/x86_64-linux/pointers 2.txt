pointers 2.swap:
	push rbp
	mov rbp, rsi
	mov rsi, rdi
	mov rax, [rbp]
	mov rcx, [rsi]
	mov [rbp], rcx
	mov [rsi], rax
	pop rbp
	ret
main:
	push rbx
	push rbp
	sub rsp, 0x18
	lea rbx, [rsp]
	mov rax, 0x01
	mov [rbx], rax
	mov rax, 0x02
	lea rbp, [rsp+0x08]
	mov [rbp], rax
	mov rdi, rbx
	mov rsi, rbp
	call :pointers 2.swap
	mov rax, [rbx]
	mov rcx, [rbp]
	sub rax, rcx
	add rsp, 0x18
	pop rbp
	pop rbx
	ret
